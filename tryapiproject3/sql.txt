create table person
(
    id          integer generated by default as identity
        primary key,
    name        varchar not null,
    age         integer not null,
    email       varchar not null,
    created_at  timestamp,
    update_at   timestamp,
    created_who varchar
);

alter table person
    owner to postgres;

create table sensor
(
    id   integer generated by default as identity
        primary key,
    name varchar not null
        unique
);

alter table sensor
    owner to postgres;

create table measument
(
    id        integer generated by default as identity
        primary key,
    value     double precision not null,
    raining   boolean          not null,
    sensor_id integer
                               references sensor
                                   on delete set null
);

alter table measument
    owner to postgres;
